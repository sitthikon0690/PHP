isset:
ตรวจสอบว่าตัวแปรถูกประกาศหรือไม่
ตรวจสอบว่าตัวแปรไม่ใช่ null
ไม่สนใจว่าตัวแปรมีค่าอะไร
<?php

$foo = null;

if (isset($foo)) {
  echo "ตัวแปร \$foo ถูกประกาศ";
} else {
  echo "ตัวแปร \$foo ไม่ได้ถูกประกาศ";
}

?>

empty:
ตรวจสอบว่าตัวแปรมีค่าว่างหรือไม่
ค่าว่าง ได้แก่ "" (string ว่าง), 0 (integer 0), 0.0 (float 0), null, false, [], array()
สนใจว่าตัวแปรมีค่าอะไร
<?php

$foo = "";

if (empty($foo)) {
  echo "ตัวแปร \$foo ว่าง";
} else {
  echo "ตัวแปร \$foo ไม่ว่าง";
}

?>
